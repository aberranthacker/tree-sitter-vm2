(source_file

--------------------------------------------------------------------------------
(source_file
  (label name: (symbol))
  (include_bin
    (directive_mnemonic)
    path: (path (string_literal))

  (comment))

================================================================================
Include binary file with offset, and length specified
================================================================================
GFX: .incbin "gfx.bin",512,256 ; Include 256 bytes, skipping 512 bytes
--------------------------------------------------------------------------------
(source_file
  (label name: (symbol))
  (include_bin
    (directive_mnemonic)
    path: (path (string_literal))
    offset: (decimal_literal)
    length: (decimal_literal))
  (comment))
